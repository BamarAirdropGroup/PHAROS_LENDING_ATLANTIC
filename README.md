# PHAROS_LENDING_ATLANTIC
PHAROS LENDING BOT ON ATLANTIC CHAIN





# Clone the repo

 git clone https://github.com/BamarAirdropGroup/PHAROS_LENDING_ATLANTIC.git && cd PHAROS_LENDING_ATLANTIC && pip install -r requirements.txt && npm install



# Add private key in accounts.txt ( NOT INCLUDE PRIVATE_KEY_1/2 ... )

 nano accounts.txt



# Add proxy if you wish 

 nano proxy.txt



# FIRST RUN THE SWAP BOT TO GET WBTC  ( 0.001 phrs or above is ok )

 python swap.py



# SECOND RUN THE LENDING BOT 

 node lending.js
